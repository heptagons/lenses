\documentclass[]{standalone}

\usepackage{../lenses}

\newcommand{\lensesHeptaLeaf}[3]{ %side,color,letter
 \lenseDD{#1}{0}{#2}{#3} %size,rot,color,letter
 \lenseDD{#1}{2}{#2}{#3}
 \begin{scope}[rotate=360/7,shift={(#1,0)}]
  \lenseEE{#1}{-1}{#2}{#3}
 \end{scope}
 \begin{scope}[rotate=360/7,shift={(#1,0)},rotate=180/7,shift={(#1,0)}]
  \lenseFF{#1}{-3}{#2}{#3}
 \end{scope}
}

\newcommand{\lensesHeptaCrown}[2]{ %side,color
 \lenseEE{#1}{4}{#2}{1} %bottom
 \begin{scope}[rotate=4*180/7,shift={(#1,0)}]
  \begin{scope}[rotate=-1*180/7,shift={(#1,0)}]
   \begin{scope}[rotate=3*180/7,shift={(#1,0)}] %\draw(0,0) node{$3$};
    \begin{scope}[xscale=-1,rotate=8*180/7]
     \lenseEE{#1}{4}{#2}{1} %top
    \end{scope}
   \end{scope}
  \end{scope}
 \end{scope}
}

\newcommand{\lensesHeptaHorns}[2]{ %side,color,rot
 \lenseDD{#1}{0}{#2}{1} %size,rot,color,letter
 \begin{scope}[rotate=2*180/7,shift={(#1,0)}] %\draw(0,0) node{$1$};
  \begin{scope}[rotate=-1*180/7] \lenseEE{#1}{0}{#2}{1} \end{scope}
  \begin{scope}[rotate=1*180/7,shift={(#1,0)}] %\draw(0,0) node{$2$};
   \begin{scope}[rotate=-3*180/7] \lenseFF{#1}{0}{#2}{1} \end{scope}
   \begin{scope}[rotate=1*180/7,shift={(#1,0)}] %\draw(0,0) node{$3$};
    \begin{scope}[rotate=1*180/7,shift={(#1,0)}] %\draw(0,0) node{$4$};
     \begin{scope}[xscale=-1,rotate=-2*180/7] \lenseEE{#1}{0}{#2}{1} \end{scope}
     \begin{scope}[rotate=1*180/7,shift={(#1,0)}] %\draw(0,0) node{$5$};
      \begin{scope}[rotate=-7*180/7] \lenseDD{#1}{0}{#2}{1} \end{scope}
     \end{scope}
    \end{scope}
   \end{scope}
  \end{scope}
 \end{scope}
}
