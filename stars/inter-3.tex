\newcommand{\moveGeneric}[3]{ %symm,angles,side
 \pgfmathsetmacro\mx{0}
 \pgfmathsetmacro\my{0}
 \pgfpathmoveto{\pgfpointxy{\mx}{\my}}
 \foreach \i in {#2} {
  \pgfmathsetmacro\tx{\mx + cos(\i*180/#1)}
  \pgfmathsetmacro\ty{\my + sin(\i*180/#1)}
  \global\let\mx\tx
  \global\let\my\ty
  \pgfpathlineto{\pgfpointxy{#3*\mx}{#3*\my}}
 }
 \pgfsetstrokecolor{blue}
 \pgfsetlinewidth{2pt}
 \pgfusepath{stroke}
 \path(\mx,\my) node{$o$};
}

\newcommand{\lineGeneric}[5]{ %symm,angles,side,color,linewidth
 \pgfmathsetmacro\mx{0}
 \pgfmathsetmacro\my{0}
 \pgfpathmoveto{\pgfpointxy{\mx}{\my}}
 \foreach \i in {#2} {
  \pgfmathsetmacro\tx{\mx + cos(\i*180/#1)}
  \pgfmathsetmacro\ty{\my + sin(\i*180/#1)}
  \global\let\mx\tx
  \global\let\my\ty
  \pgfpathlineto{\pgfpointxy{#3*\mx}{#3*\my}}
 }
 \pgfsetstrokecolor{#4}
 \pgfsetlinewidth{#5}
 \pgfusepath{stroke}
}
